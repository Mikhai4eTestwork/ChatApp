Процесс сборки

1) docker-compose -d --build #запускаем сервисы
2) cd front #директория, в которой лежит фронт
3) npm install  #инсталлим модули
3) npm vue serve #запускаем локально
5) npm run build #сбилдить продуктовую версию

#проект тестировался только на локальной машине, весь фронт завязан на локальные хосты
#эндпоинты определены в файле /front/src/modules/endpoint.js
